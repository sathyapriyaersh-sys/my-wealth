<script>

let totalIncome = 0;
let totalExpense = 0;
let debtAmount = 0;
let dueDate = null;

function updateDashboard(){
    totalIncome = parseFloat(document.getElementById("income").value) || 0;
    document.getElementById("totalIncome").innerText = totalIncome;
}

function addDaily(){
    let dIncome = parseFloat(document.getElementById("dailyIncome").value) || 0;
    let dExpense = parseFloat(document.getElementById("dailyExpense").value) || 0;

    totalIncome += dIncome;
    totalExpense += dExpense;

    let savings = totalIncome - totalExpense;

    document.getElementById("totalIncome").innerText = totalIncome;
    document.getElementById("totalExpense").innerText = totalExpense;
    document.getElementById("totalSavings").innerText = savings;

    if(debtAmount > 0 && dueDate){
        let today = new Date();
        let timeDiff = new Date(dueDate) - today;
        let daysLeft = Math.ceil(timeDiff / (1000*60*60*24));

        if(daysLeft > 0){
            let requiredPerDay = debtAmount / daysLeft;

            if(dIncome - dExpense >= requiredPerDay){
                document.getElementById("appreciation").innerHTML =
                "<span class='success'>Great job! You saved enough today towards your debt.</span>";
            }else{
                document.getElementById("appreciation").innerHTML =
                "<span class='warning'>You need to save more daily to clear debt on time.</span>";
            }
        }
    }
}

function calculateDebt(){
    debtAmount = parseFloat(document.getElementById("debtAmount").value) || 0;
    dueDate = document.getElementById("dueDate").value;

    document.getElementById("totalDebt").innerText = debtAmount;

    let today = new Date();
    let timeDiff = new Date(dueDate) - today;
    let daysLeft = Math.ceil(timeDiff / (1000*60*60*24));

    if(daysLeft <= 0){
        document.getElementById("debtInfo").innerHTML =
        "<span class='danger'>Due date has passed or is today.</span>";
        return;
    }

    let requiredPerDay = (debtAmount / daysLeft).toFixed(2);

    document.getElementById("debtInfo").innerHTML =
    "You must save <b>₹" + requiredPerDay + "</b> per day to clear debt in " + daysLeft + " days.";

    if(daysLeft == 1){
        alert("Reminder: Your debt is due tomorrow. Please arrange payment.");
    }
}
